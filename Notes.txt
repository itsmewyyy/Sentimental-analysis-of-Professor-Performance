
--ACTIVATING VIRTUAL ENVIRONMENT--
activate the virtual env first
.\venv\Scripts\activate

if error, delete venv and reinstall
python -m venv venv

--INSTALL REQS--
to install dependencies
pip install -r requirements.txt


--DATABASE MIGRATE--

Create a username and password for mysql
CREATE USER 'admin'@'localhost' IDENTIFIED BY 'plpsetadmin';

create a database named "plpset" 

run these commands:
python manage.py migrate 

create a superuser same as the system
python manage.py createsuperuser

username: plpsetadmin
pw: plpset@admin_1234


--RUNNING THE SYSTEM--
Use 2 terminals, one for backend (django) and one for frontend (vue)

DJANGO
Ensure that venv is running
cd djangoSet
python manage.py runserver

VUE 
cd djangoSet
cd frontendSet
npm run dev
